// ───────────────────────────────────────────────────────────── // SharePal Payout Tracker — Google Apps Script // Paste this entire file into Extensions > Apps Script > Code.gs // Deploy > New deployment > Web app // Execute as: Me // Who has access: Anyone // ───────────────────────────────────────────────────────────── const SHEET_NAME_PARTNERS = "All"; const SHEET_NAME_STATUS = "Payout_Status"; const SHEET_NAME_USERS = "Users"; // ── USERS — edit this list to add/remove users ──────────────── // Roles: "admin" (full access + contact info) | "finance" (can mark processed) | "viewer" (read-only) const USERS = [ { username: "Deep", pin: "1234", role: "admin" }, { username: "Finance", pin: "9999", role: "finance" }, // Add more users below: // { username: "Rahul", pin: "5678", role: "viewer" }, ]; // Everything goes through doGet to avoid CORS issues function doGet(e) { const p = e && e.parameter ? e.parameter : {}; const action = p.action || ""; const username = (p.username || "").trim(); const pin = (p.pin || "").trim(); let result; try { if (action === "ping") { result = { ok: true }; } else if (action === "getUsers") { // Returns only usernames — no PINs or roles exposed result = { ok: true, users: USERS.map(u => u.username) }; } else if (action === "verifyPin") { result = verifyUser(username, pin); } else if (action === "getAll") { const user = getUser(username, pin); if (!user) { result = { error: "Unauthorised" }; } else { result = getAll(); result.role = user.role; result.username = user.username; } } else if (action === "markProcessed") { const user = getUser(username, pin); if (!user) { result = { error: "Unauthorised" }; } else if (user.role === "viewer") { result = { error: "No permission" }; } else { result = markProcessed(p.payoutKey, p.transactionId, p.note || "", user.username); } } else if (action === "undoPayout") { const user = getUser(username, pin); if (!user || user.role !== "admin") { result = { error: "Admin only" }; } else { result = undoPayout(p.payoutKey); } } else if (action === "getAgreementData") { const user = getUser(username, pin); if (!user || user.role !== "admin") { result = { error: "Admin only" }; } else { result = getAgreementData(p.partnerId); } } else if (action === "getPartnerList") { const user = getUser(username, pin); if (!user || user.role !== "admin") { result = { error: "Admin only" }; } else { result = getPartnerList(); } } else { result = { error: "Unknown action: " + action }; } } catch (err) { result = { error: err.message }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ── AUTH ────────────────────────────────────────────────────── function verifyUser(username, pin) { if (!username) return { ok: false, error: "Enter your name" }; const user = USERS.find(u => u.username.toLowerCase() === username.toLowerCase() && u.pin === pin); if (user) return { ok: true, role: user.role, username: user.username }; return { ok: false, error: "Wrong name or PIN" }; } function getUser(username, pin) { return USERS.find(u => u.username.toLowerCase() === username.toLowerCase() && u.pin === pin) || null; } // ── ENSURE STATUS SHEET ─────────────────────────────────────── function ensureStatusSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sh = ss.getSheetByName(SHEET_NAME_STATUS); if (!sh) { sh = ss.insertSheet(SHEET_NAME_STATUS); sh.appendRow([ "PayoutKey", "PartnerID", "PartnerName", "PayoutNum", "DueDate", "Amount", "Status", "TransactionID", "MarkedBy", "MarkedAt", "Note" ]); sh.setFrozenRows(1); sh.getRange(1, 1, 1, 11).setFontWeight("bold"); } return sh; } // ── PARSE PARTNER DATA ──────────────────────────────────────── function parsePartners() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sh = ss.getSheetByName(SHEET_NAME_PARTNERS); const rows = sh.getDataRange().getValues(); const partners = []; for (let i = 1; i < rows.length; i++) { const r = rows[i]; if (!r[0]) continue; // Use AFP ID if present, otherwise generate a stable fallback from email+name const afpId = String(r[1]).trim(); const stableId = afpId || ("NO-ID-" + String(r[2]).trim().replace(/[^a-zA-Z0-9]/g,"").toUpperCase().slice(0,12)); partners.push({ name: String(r[0]), id: stableId, afpId: afpId, // original — may be empty email: String(r[2]), phone: String(r[3]), invested: parseAmount(r[4]), model: String(r[5]), tenure: parseInt(String(r[6])), totalPayout:parseAmount(r[7]), monthly: parseAmount(r[8]), investDate: parseDate(r[9]), payType: String(r[10]), txnId: String(r[11] || ""), // ── Agreement fields ── pan: String(r[12] || ""), aadhaar: String(r[13] || ""), address: String(r[14] || ""), assetNames: String(r[18] || ""), assetSerials: String(r[19] || ""), assetValues: String(r[20] || ""), }); } return partners; } function parseAmount(val) { if (typeof val === "number") return val; return parseFloat(String(val).replace(/[₹,\s]/g, "")) || 0; } function parseDate(val) { if (val instanceof Date) return val; if (!val) return null; const s = String(val).trim(); const months = {Jan:0,Feb:1,Mar:2,Apr:3,May:4,Jun:5,Jul:6,Aug:7,Sep:8,Oct:9,Nov:10,Dec:11}; const m = s.match(/(\d{1,2})-([A-Za-z]{3})-(\d{4})/); if (m) return new Date(+m[3], months[m[2]], +m[1]); return new Date(s); } // ── BUILD PAYOUT SCHEDULE ───────────────────────────────────── function buildPayoutSchedule(partners) { const payouts = []; partners.forEach(p => { if (!p.investDate) return; const base = new Date(p.investDate); const isFlat = p.payType && p.payType.toLowerCase().includes("flat"); const isYearly = p.model && p.model.toLowerCase().includes("yearly"); if (isYearly) { // Yearly model: one payout every 12 months // Number of yearly payouts = tenure / 12 const numPayouts = Math.round(p.tenure / 12); for (let i = 1; i <= numPayouts; i++) { const due = new Date(base); due.setMonth(due.getMonth() + (i * 12)); const isLast = i === numPayouts; const amount = Math.round(p.monthly) + (isFlat && isLast ? Math.round(p.invested) : 0); payouts.push({ key: p.id + "-" + i, partnerId: p.id, afpId: p.afpId || "", partnerName: p.name, payoutNum: i, tenure: p.tenure, totalPayouts: numPayouts, dueDate: due.toISOString(), amount, isLastPayout: isLast, includesPrincipal: isFlat && isLast, payoutModel: "Yearly", }); } } else { // Monthly model: one payout every month for (let i = 1; i <= p.tenure; i++) { const due = new Date(base); due.setMonth(due.getMonth() + i); const isLast = i === p.tenure; const amount = Math.round(p.monthly) + (isFlat && isLast ? Math.round(p.invested) : 0); payouts.push({ key: p.id + "-" + i, partnerId: p.id, afpId: p.afpId || "", partnerName: p.name, payoutNum: i, tenure: p.tenure, totalPayouts: p.tenure, dueDate: due.toISOString(), amount, isLastPayout: isLast, includesPrincipal: isFlat && isLast, payoutModel: "Monthly", }); } } }); return payouts; } // ── READ STATUS MAP ─────────────────────────────────────────── function readStatusMap() { const sh = ensureStatusSheet(); const rows = sh.getDataRange().getValues(); const map = {}; for (let i = 1; i < rows.length; i++) { const r = rows[i]; if (!r[0]) continue; map[String(r[0])] = { status: String(r[6]), transactionId: String(r[7]), markedBy: String(r[8]), markedAt: String(r[9]), note: String(r[10]), rowIndex: i + 1, }; } return map; } // ── GET ALL ─────────────────────────────────────────────────── function getAll() { const partners = parsePartners(); const payouts = buildPayoutSchedule(partners); const statusMap = readStatusMap(); payouts.forEach(p => { const s = statusMap[p.key]; if (s) { p.status = s.status; p.transactionId = s.transactionId; p.markedBy = s.markedBy; p.markedAt = s.markedAt; p.note = s.note; } }); return { ok: true, partners, payouts }; } // ── MARK PROCESSED ──────────────────────────────────────────── function markProcessed(payoutKey, transactionId, note, markedBy) { if (!payoutKey) return { error: "payoutKey required" }; if (!transactionId || !transactionId.trim()) return { error: "Transaction ID is mandatory" }; const sh = ensureStatusSheet(); const statusMap = readStatusMap(); const partners = parsePartners(); const payouts = buildPayoutSchedule(partners); const payout = payouts.find(p => p.key === payoutKey); if (!payout) return { error: "Payout not found: " + payoutKey }; const now = new Date().toISOString(); if (statusMap[payoutKey]) { const row = statusMap[payoutKey].rowIndex; sh.getRange(row, 7).setValue("Processed"); sh.getRange(row, 8).setValue(transactionId.trim()); sh.getRange(row, 9).setValue(markedBy); sh.getRange(row, 10).setValue(now); sh.getRange(row, 11).setValue(note || ""); } else { sh.appendRow([ payoutKey, payout.partnerId, payout.partnerName, payout.payoutNum, new Date(payout.dueDate), payout.amount, "Processed", transactionId.trim(), markedBy, now, note || "", ]); } return { ok: true }; } // ── UNDO PAYOUT (admin only) ────────────────────────────────── function undoPayout(payoutKey) { const sh = ensureStatusSheet(); const statusMap = readStatusMap(); if (!statusMap[payoutKey]) return { error: "Not found" }; const row = statusMap[payoutKey].rowIndex; sh.getRange(row, 7).setValue("Pending"); sh.getRange(row, 8).setValue(""); sh.getRange(row, 9).setValue(""); sh.getRange(row, 10).setValue(new Date().toISOString()); sh.getRange(row, 11).setValue("Undone by admin"); return { ok: true }; } // ── AGREEMENT DATA ──────────────────────────────────────────── // Maps a partner record to agreement.html field IDs function getAgreementData(partnerId) { if (!partnerId) return { error: "partnerId required" }; const partners = parsePartners(); const p = partners.find(x => x.id === partnerId); if (!p) return { error: "Partner not found" }; // Determine agreement type from Payout Model + Payout Type const isFlat = p.payType && p.payType.toLowerCase().includes("flat"); const isYearly = p.model && p.model.toLowerCase().includes("yearly"); let agreeType; if (isYearly && isFlat) agreeType = "flat_yearly"; else if (!isYearly && isFlat) agreeType = "flat_monthly"; else agreeType = "emi"; // Split comma-separated asset fields const splitCsv = s => String(s||"").split(",").map(x=>x.trim()).filter(x=>x); const names = splitCsv(p.assetNames); const serials = splitCsv(p.assetSerials); const values = splitCsv(p.assetValues).map(v => parseAmount(v)); const assets = names.map((name, i) => ({ name: name, serial: serials[i] || "", value: values[i] !== undefined ? values[i] : "", })); // Warn if asset values don't sum to invested amount let assetValueWarning = ""; if (values.length > 0) { const sum = values.reduce((a,b)=>a+b, 0); if (Math.abs(sum - p.invested) > 1) { assetValueWarning = "Asset values sum to " + sum + " but Amount Invested is " + p.invested + ". Please review."; } } // Effective date as YYYY-MM-DD for const effectiveDate = p.investDate ? Utilities.formatDate(p.investDate, "Etc/GMT", "yyyy-MM-dd") : ""; return { ok: true, data: { // Step 1 — Lender Info lender_name: p.name, lender_pan: p.pan, lender_aadhaar: p.aadhaar, lender_address: p.address, lender_email: p.email, lender_mobile: p.phone, // Step 2 — Loan Terms loan_amount: p.invested, effective_date: effectiveDate, tenure: p.tenure, monthly_payout: p.monthly, agreeType: agreeType, // Step 3 — Transaction (txn date/amount = investment date/amount) txn_date: effectiveDate, txn_amount: p.invested, txn_id: p.txnId, // Step 4 — Assets assets: assets, // Metadata partnerId: p.id, afpId: p.afpId, assetValueWarning: assetValueWarning, } }; } // ── PARTNER LIST (for manual picker in agreement.html) ──────── function getPartnerList() { const partners = parsePartners(); return { ok: true, partners: partners.map(p => ({ id: p.id, afpId: p.afpId, name: p.name, invested: p.invested, })), }; }
